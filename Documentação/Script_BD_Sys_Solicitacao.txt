CREATE TABLE [dbo].[TB_LOGIN] (
    [I_COD_LOGIN] INT          IDENTITY (1, 1) NOT NULL,
    [S_PW_LOGIN]  VARCHAR (32) NOT NULL,
    [S_UNM_LOGIN] VARCHAR (15) NOT NULL,
    PRIMARY KEY CLUSTERED ([I_COD_LOGIN] ASC)
);

CREATE TABLE [dbo].[TB_CLIENTE] 
(
    [I_COD_CLIENTE]   INT          IDENTITY (1, 1) NOT NULL,
    [S_NM_CLIENTE]  VARCHAR(35)  NOT NULL,
    [S_EMAIL_CLIENTE] VARCHAR(35)  NOT NULL,
    [S_RG_CLIENTE]    VARCHAR(9)   NOT NULL,
    [S_CPF_CLIENTE]   VARCHAR(11)  NOT NULL,
    PRIMARY KEY CLUSTERED ([I_COD_CLIENTE] ASC)
);

CREATE TABLE [dbo].[TB_FUNCIONARIO] 
(
    [I_COD_FUNCIONARIO]     INT          IDENTITY (1, 1) NOT NULL,
    [S_NM_FUNCIONARIO]    VARCHAR(35)     NOT NULL,
    [S_CPF_FUNCIONARIO]     VARCHAR(11)   NOT NULL,
    [S_END_FUNCIONARIO]     VARCHAR(40)   NOT NULL,
    [S_BAI_FUNCIONARIO]     VARCHAR(30)   NOT NULL,
    [S_CID_FUNCIONARIO]     VARCHAR(30)   NOT NULL,
    [S_UF_FUNCIONARIO]     VARCHAR(2)     NOT NULL,
    [S_CEP_FUNCIONARIO]     VARCHAR(30)   NOT NULL,
    [S_EMAIL_FUNCIONARIO]     VARCHAR(40)  NOT NULL,
    [S_CEL_FUNCIONARIO]      VARCHAR(11)  NOT NULL,
    [DT_NASC_FUNCIONARIO]      DATETIME   NOT NULL,
    PRIMARY KEY CLUSTERED ([I_COD_FUNCIONARIO] ASC)
);

CREATE TABLE [dbo].[TB_GENERO] 
(
    [I_COD_GENERO]         INT           IDENTITY (1, 1) NOT NULL,
    [S_NM_GENERO]        VARCHAR (20)  NOT NULL,
    PRIMARY KEY CLUSTERED ([I_COD_GENERO] ASC)
);

CREATE TABLE [dbo].[TB_ARTISTA] (
    [I_COD_ARTISTA]    INT          IDENTITY (1, 1) NOT NULL,
    [I_COD_GENERO]     INT NOT NULL,
    [S_NM_ARTISTA]     VARCHAR (35) NOT NULL,
    PRIMARY KEY CLUSTERED ([I_COD_ARTISTA] ASC),
    FOREIGN KEY ([I_COD_GENERO]) REFERENCES [dbo].[TB_GENERO] ([I_COD_GENERO])
);

CREATE TABLE [dbo].[TB_ESPETACULO] 
(
    [I_COD_ESPETACULO]      INT           IDENTITY (1, 1) NOT NULL,
    [I_COD_ARTISTA]         INT           NOT NULL,
    [S_NM_ESPETACULO]     VARCHAR (40) NOT NULL,
    [DT_DTHR_ESPETACULO]    DATETIME      NOT NULL,  
    PRIMARY KEY CLUSTERED ([I_COD_ESPETACULO] ASC),
    FOREIGN KEY ([I_COD_ARTISTA]) REFERENCES [dbo].[TB_ARTISTA] ([I_COD_ARTISTA])
);

CREATE TABLE [dbo].[TB_PEDIDO] 
(
    [I_COD_PEDIDO]       INT          IDENTITY (1, 1) NOT NULL,
    [I_COD_FUNCIONARIO]  INT          NOT NULL,
    [I_COD_CLIENTE]      INT          NOT NULL,
    [DT_DTHR_PEDIDO]  DATETIME NOT NULL,
    PRIMARY KEY CLUSTERED ([I_COD_PEDIDO] ASC),
    FOREIGN KEY ([I_COD_FUNCIONARIO]) REFERENCES [dbo].[TB_FUNCIONARIO] ([I_COD_FUNCIONARIO]),
    FOREIGN KEY ([I_COD_CLIENTE]) REFERENCES [dbo].[TB_CLIENTE] ([I_COD_CLIENTE])
);

CREATE TABLE [dbo].[TB_ITEM_PEDIDO] 
(
    [I_COD_ITEM_PEDIDO]   INT          IDENTITY (1, 1) NOT NULL,
    [I_COD_PEDIDO]        INT          NOT NULL,
    [I_COD_ESPETACULO]    INT          NOT NULL,
    [D_VLR_ITEM_PEDIDO]   FLOAT        NOT NULL,
    [D_DESC_ITEM_PEDIDO]  FLOAT        NOT NULL,
    [I_QUANT_ITEM_PEDIDO] INT          NOT NULL,
    PRIMARY KEY CLUSTERED ([I_COD_ITEM_PEDIDO] ASC),
    FOREIGN KEY ([I_COD_PEDIDO]) REFERENCES [dbo].[TB_PEDIDO] ([I_COD_PEDIDO]),
    FOREIGN KEY ([I_COD_ESPETACULO]) REFERENCES [dbo].[TB_ESPETACULO] ([I_COD_ESPETACULO])
);
